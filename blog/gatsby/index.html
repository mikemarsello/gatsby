Using images in Gatsby

Static Query

Use a static query if you need a query in a component that is NOT in /pages

import { StaticQuery, graphql } from 'gatsby'
import Img from 'gatsby-image'

<StaticQuery
  query={graphql`
    query {
      img1: file(relativePath: { eq: "IMG_1455.JPG" }) {
        ...fluidImage
      }
      img2: file(relativePath: { eq: "IMG_0358.JPG" }) {
        ...fluidImage
      }
      img3: file(relativePath: { eq: "IMG_1541.JPG" }) {
        ...fluidImage
      }
    }
  `}

render={data => (
  <div>
    <Img fluid={data.img1.childImageSharp.fluid} />
    <Img fluid={data.img2.childImageSharp.fluid} />
    <Img fluid={data.img3.childImageSharp.fluid} />
  </div>
)}

export const fluidImage = graphql`
  fragment fluidImage on File {
    childImageSharp {
      fluid(maxWidth: 1000) {
        ...GatsbyImageSharpFluid
      }
    }
  }
`
